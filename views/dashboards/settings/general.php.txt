    if(isset($post_data['save_contact_settings'])) $form_name = 'contact';

    if ($form_name) {
        $save_ok = true;
        
        // Handle image upload only for the branding form
        if ($form_name === 'branding' && isset($_FILES['hero_image']) && $_FILES['hero_image']['error'] === UPLOAD_ERR_OK) {
            if (!Settings::handleImageUpload('hero_image_path', $_FILES['hero_image'], $user_id)) {
                $save_ok = false;
                $_SESSION['flash_message'] = ['type' => 'error', 'message' => 'Failed to upload hero image.'];
            }
        }
        
        if ($save_ok) {
            if (Settings::save($post_data, $user_id)) {
                $_SESSION['flash_message'] = ['type' => 'success', 'message' => 'Settings saved successfully!'];
            } else {
                $_SESSION['flash_message'] = ['type' => 'error', 'message' => 'Failed to save settings to the database.'];
            }
        }
    }
    
    header("Location: " . $_SERVER['REQUEST_URI']);
    exit;
}
